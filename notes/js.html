动态更改粘性定位 ​

js
// 获取目标元素
const element = document.getElementById('yourElementId');

// 创建一个 IntersectionObserver 实例
const observer = new IntersectionObserver(entries => {
  entries.forEach(entry => {
    if (entry.intersectionRatio < 0.5) {
      element.classList.add('sticky-top'); // 添加粘性定位的类名
    } else {
      element.classList.remove('sticky-top'); // 移除粘性定位的类名
    }
  });
}, { threshold: 0.5 });

// 开始观察目标元素
observer.observe(element);

距离浏览器窗口底部的距离 ​

js
// 获取<div>元素
var divElement = document.getElementById('your-div-id');

// 获取<div>元素底部相对于文档顶部的距离
var divBottom = divElement.getBoundingClientRect().bottom;

// 获取视口（浏览器窗口）的高度
var viewportHeight = window.innerHeight;

// 计算<div>元素底部距离浏览器窗口底部的距离
var distanceToBottom = viewportHeight - divBottom;

console.log('距离浏览器窗口底部的距离为：' + distanceToBottom + ' 像素');
