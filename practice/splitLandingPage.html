3 main.ts: ​

ts
import './style.scss'
// 选择容器元素
const container = document.querySelector('body')!;
// 当容器大小发生变化时执行的函数
const handleResize = () => {
document.querySelectorAll<HTMLDivElement>('.title').forEach((item)=>{
    if(container.clientWidth<850){
        item.style.fontSize= '2rem'
        item.style.top= '30%'
    }
    else{
        item.style.fontSize='' 
        item.style.top= ''
    }
})
};
// 创建 ResizeObserver 实例，监测容器大小的变化
const resizeObserver = new ResizeObserver(handleResize);
resizeObserver.observe(container);
