3 main.ts: ​

ts
import './style.scss'

const container = document.querySelector('.top')!;
const image = document.querySelector<HTMLImageElement>('.top img')!;
const handleResize = () => {
    const containerWidth = container.clientWidth;
    const containerHeight = container.clientHeight;
    const imageWidth = image.naturalWidth;
    const imageHeight = image.naturalHeight;
    const containerAspectRatio = containerWidth / containerHeight;
    const imageAspectRatio = imageWidth / imageHeight;

    if (containerAspectRatio > imageAspectRatio) {
        container.classList.add('fill');
        container.classList.remove('contain');
    } else {
        container.classList.add('contain');
        container.classList.remove('fill');
    }
};
const resizeObserver = new ResizeObserver(handleResize);
resizeObserver.observe(container);
image.onload = () => {
    handleResize()
};




let br = [...document.querySelector('ul')!.children]
let br_windows = [...document.querySelector('.br_windows')!.children].map(element => element.children);
let mains = [document.querySelector('.top')!, document.querySelector('.bottom')!]
mains.forEach((main) => {
    main.addEventListener("click", () => {
        br_windows.forEach((element, index) => {
            element[0].classList.remove("show");
            br[index].classList.remove("show");
        })
    })
})
br.forEach((element, i) => {
    element.addEventListener('click', (event) => {
        event.stopPropagation()
        let temp = false;
        br_windows.forEach((element, j) => {
            if (i != j && element[0].classList.contains("show")) {
                temp = true
                element[0].classList.remove("show");
                br[j].classList.remove("show");
            }
        })
        if (temp) {
            setTimeout(() => {
                br_windows[i][0].classList.toggle("show")
                br[i].classList.toggle("show")
            }, 300)
        }
        else {
            br_windows[i][0].classList.toggle("show")
            br[i].classList.toggle("show")
        }
    })
});
