3 main.ts: ​

ts
import './style.scss'
let divs = document.querySelectorAll<HTMLDivElement>(".content")

let num = 0;
let distanceToBottom = () => {
    let divElement = divs[num];
    // 获取<div>元素底部相对于文档顶部的距离
    let divBottom = divElement.getBoundingClientRect().top;
    // 获取视口（浏览器窗口）的高度
    let viewportHeight = window.innerHeight;
    // 计算<div>元素底部距离浏览器窗口底部的距离
    return viewportHeight - divBottom
};
let switchClass = function () {
    // console.log(distanceToBottom());
    // console.log(num);
    while (distanceToBottom() > 100) {
        divs[num].classList.add('show')
        num++
        if (num > divs.length - 1) {
            num = divs.length - 1;
            break
        }
    }
    while (distanceToBottom() < 140) {
        divs[num].classList.remove('show')
        num--
        if (num < 0) {
            num = 0;
            break
        }
    }
}
setTimeout(() => {
    switchClass();
    window.addEventListener('scroll', switchClass); 
}, 300);
